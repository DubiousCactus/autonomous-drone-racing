        def apply_motion_blur(self, img, amount=0.5):
            if self.no_blur: return cv_img
            if amount <= 0.3:
                size = 3
            elif amount <= 0.7:
                size = 5
            else:
                size = 9
            kernel = np.identity(size) / size
            return cv2.filter2D(img, -1, kernel)
