\def\angle{90}
\def\radius{3}
\def\cyclelist{{"grey5","grey4","grey3","grey2","grey1"}}
\newcount\cyclecount \cyclecount=-1
\newcount\ind \ind=-1
\begin{tikzpicture}[scale=1]
 \begin{scope}[xshift=0cm]
  \foreach \percent/\name in {
      56/\text EO+RS,
      10/Technology,
      27/Scientific,
      6/Communications,
      1/Novel Applications
    } {
    \ifx\percent\empty\else % If \percent is empty, do nothing
    \global\advance\cyclecount by 1 % Advance cyclecount
    \global\advance\ind by 1 % Advance list index
    \ifnum4<\cyclecount % If cyclecount is larger than list
    \global\cyclecount=0 % reset cyclecount and
    \global\ind=0 % reset list index
    \fi
    \draw[white,line width=1.2mm](0,0)--++(\angle:\radius*1.01);
    \pgfmathparse{\cyclelist[\the\ind]} % Get color from list
    \edef\color{\pgfmathresult} % and store as \color
    % Draw angle and set labels
    \draw[fill={\color!80},draw={\color!80}] (0,0) -- (\angle:\radius)
   arc (\angle:\angle-\percent*3.6:\radius) -- cycle;  
   \ifnum6>\percent
   \node at (\angle-0.5*\percent*3.6:1.1*\radius) {\percent\,\%};
   %\node[pin=\angle-0.5*\percent*3.6:\small\name] at (\angle-0.5*\percent*3.6:1.2*\radius) {};
   \else
    \node at (\angle-0.5*\percent*3.6:0.7*\radius) {\percent\,\%};
    %\node[pin=\angle-0.5*\percent*3.6:\small\name] at (\angle-0.5*\percent*3.6:\radius) {};
    \fi

   \pgfmathparse{\angle-\percent*3.6} % Advance angle
   \xdef\angle{\pgfmathresult} % and store in \angle
   \fi
};
\draw[white,line width=0.6mm](0,0)--++(\angle:\radius*1.01);
\fill[white](0,0)circle [radius=\radius*0.4];

\node [label,align=center] {\footnotesize Historical \\[-4pt]\footnotesize 2014 - 2018};

\end{scope}
